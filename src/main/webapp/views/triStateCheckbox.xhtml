The TriStateCheckbox add a new state for a common SelectBooleanCheckbox. The value of triStateCheckbox are mapped to 0,1,2 string values, the component also can be customized with css icon classes.

This example show triStateCheckbox in action.
<p:growl id="msg" /> <h:outputText value="Basic Usage: " /> <pe:triStateCheckbox value="\#{triStateCheckBoxController.value1}" /> <h:outputText value="Ajax Behavior: " /> <pe:triStateCheckbox value="\#{triStateCheckBoxController.value2}"> <p:ajax update="display" listener="\#{triStateCheckBoxController.addMessage}"/> </pe:triStateCheckbox> <h:outputText value="Customs Icons: " /> <pe:triStateCheckbox value="\#{triStateCheckBoxController.value1}" stateTwoIcon="ui-icon-plus" stateThreeIcon="ui-icon-minus"/> <p:commandButton value="Submit" update="display" oncomplete="dlg.show()" /> <p:dialog header="Selected Values" showEffect="fade" hideEffect="fade" widgetVar="dlg"> <h:panelGrid columns="1" id="display"> <h:outputText value="Value 1: \#{triStateCheckBoxController.value1}" /> <h:outputText value="Value 2: \#{triStateCheckBoxController.value2}" /> <h:outputText value="Value 3: \#{triStateCheckBoxController.value3}" /> </h:panelGrid> </p:dialog> @ManagedBean @ViewScoped public class TriStateCheckBoxController implements Serializable { private String value1; private String value2; private String value3; public TriStateCheckBoxController() { value1 = "2"; } public String getValue1() { return value1; } public void setValue1(String value1) { this.value1 = value1; } public String getValue2() { return value2; } public String getValue3() { return value3; } public void setValue3(String value3) { this.value3 = value3; } public void setValue2(String value2) { this.value2 = value2; } public void addMessage() { String checkBoxValue = "State is = " + value2; FacesContext.getCurrentInstance().addMessage(null, new FacesMessage(checkBoxValue)); } }
