function showContent() { $('#textArea').val(PF('timelineWdgt').getSelectedEvent().content); } function add() { var range = PF('timelineWdgt').getVisibleRange(); var start = new Date((range.start.valueOf() + range.end.valueOf()) / 2); var content = $('#textArea').val(); PF('timelineWdgt').addEvent({ 'start': start, 'content': content }); var count = PF('timelineWdgt').getNumberOfEvents(); PF('timelineWdgt').setSelection(count - 1); } function change() { var index = PF('timelineWdgt').getSelectedIndex(); if (index == -1) { // no event was selected return; } if (!PF('timelineWdgt').isEditable(index)) { // only editable events can be changed PF('infoTipWdgt').show(); return; } var content = $('#textArea').val(); PF('timelineWdgt').changeEvent(index, { 'content': content // Note: start, end, and group can be added here too. }); } function remove() { var index = PF('timelineWdgt').getSelectedIndex(); if (index == -1) { // no event was selected return; } if (!PF('timelineWdgt').isEditable(index)) { // only editable events can be deleted PF('infoTipWdgt').show(); return; } PF('confirmWdgt').show(); } function removeAfterConfirm() { var index = PF('timelineWdgt').getSelectedIndex(); PF('timelineWdgt').deleteEvent(index); $('#textArea').val(''); }
